#Regression tests for existing functionality, to ensure nothing was broken.

is

</artworkPage>

!!test
  <artworkPage> with recursive parsing
!!input
<artworkPage>
this ''is'' a '''test'''
</artworkPage>
!! result
<div class="artworkPage">
<p>this <i>is</i> a <b>test</b>
</p>
</div>

!!end


!!test
  <artworkPage> with leading whitespace
!!input
<artworkPage>

   test

</artworkPage>
!! result
<div class="artworkPage">
<p><br />
&#160;&#160;&#160;test<br />
</p>
</div>

!!end
